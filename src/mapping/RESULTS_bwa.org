library(ggplot2)
library(reshape2)

  makeDataFrame <- function(df) {
    data <- df[-1,-1]
    colnames(data) <- c("mapped", "subst", "ins", "del", 
                        "indel_100bp", "indel_read", 
                        "subst_100bp", "subst_read", "cov")
    data$reads <- c("before correction",
                    "after correction (trimming uncorrected ends)",
                    "after correction (keeping uncorrected ends)")
    data$indel_100bp <- as.numeric(data$indel_100bp)
    data$subst_100bp <- as.numeric(data$subst_100bp)
    melt(data)
  }
  
  makePlot <- function(df) {
    data <- makeDataFrame(df)
    ggplot(data=data, aes(x=data$variable, y=data$value, 
                          fill=data$reads, group=data$reads)) + 
      geom_bar(stat="identity", position="dodge")
  }
